Retrieved reviews for two Sephora brush products: SEPHORA COLLECTION PRO Featherweight Powder Brush #91 and Sephora Collection Skin Perfecting Brush Set to analyze customer feedback.
```
# Get reviews for both Sephora brush products
reviews_brush_91 = magento_review_server_get_product_reviews("B01N77Y4ZH")
reviews_brush_set = magento_review_server_get_product_reviews("B08YP1ZGNJ")

# Combine and format the reviews
all_reviews = []

if reviews_brush_91:
    for review in reviews_brush_91:
        all_reviews.append({
            'product': 'SEPHORA COLLECTION PRO Featherweight Powder Brush #91',
            'reviewer': review.get('nickname', 'Anonymous'),
            'rating': review.get('rating', 'N/A'),
            'title': review.get('title', ''),
            'detail': review.get('detail', '')
        })

if reviews_brush_set:
    for review in reviews_brush_set:
        all_reviews.append({
            'product': 'Sephora Collection Skin Perfecting Brush Set',
            'reviewer': review.get('nickname', 'Anonymous'),
            'rating': review.get('rating', 'N/A'),
            'title': review.get('title', ''),
            'detail': review.get('detail', '')
        })

# Format and send the results to the user
if all_reviews:
    message = "Here's what customers say about Sephora brushes:\n\n"
    for review in all_reviews:
        message += f"**Product:** {review['product']}\n"
        message += f"**Reviewer:** {review['reviewer']}\n"
        message += f"**Rating:** {review['rating']}/5\n"
        message += f"**Title:** {review['title']}\n"
        message += f"**Review:** {review['detail']}\n"
        message += "---\n"
    send_msg_to_user(message)
else:
    send_msg_to_user("No reviews found for Sephora brush products yet.")
```
